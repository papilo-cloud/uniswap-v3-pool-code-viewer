Immutables & Slot0 — The packed price struct
uint160 · int24 · uint16 · uint8 · bool
    // ─────────────── Immutables (set once in constructor) ─────────────────
    address public immutable override factory;
    address public immutable override token0;
    address public immutable override token1;
    uint24  public immutable override fee;
    int24   public immutable override tickSpacing;
    uint128 public immutable override maxLiquidityPerTick;

    // ─────────────── Slot0: single 32-byte storage slot ───────────────────
    /// @dev All fields pack into exactly one 256-bit storage word.
    /// Reading price, tick, and the reentrancy lock costs ONE cold SLOAD (~2100 gas).
    struct Slot0 {
        // Q64.96 fixed-point: actual price = (sqrtPriceX96 / 2^96)^2
        // Range covers ~1.5×10^-29 to ~3.4×10^38 (enough for any token pair)
        uint160 sqrtPriceX96;  // 160 bits  — current sqrt(price) * 2^96

        // floor(log_1.0001(price)) — integer tick for the current price.
        // int24 covers ±8,388,607 but pool restricts to ±887,272.
        int24   tick;           // 24 bits   — current tick index

        // Index into the observations circular buffer (wraps at cardinality).
        uint16  observationIndex;        // 16 bits

        // How many observations slots are currently active (max 65535).
        uint16  observationCardinality;   // 16 bits

        // Next cardinality to grow to when an observation is written.
        uint16  observationCardinalityNext; // 16 bits

        // Protocol fee: 4 low bits = fee on token0, 4 high bits = fee on token1.
        // Value N means the protocol takes 1/N of LP fees. 0 = disabled.
        uint8   feeProtocol;    // 8 bits

        // Reentrancy lock — true means a call is in progress.
        // Stored in slot0 so the same SLOAD that reads price also reads the lock.
        bool    unlocked;       // 8 bits  (total = 160+24+16+16+16+8+8 = 248 bits ≤ 256)
    }
    Slot0 public override slot0;

    // Mutex: reverts if unlocked == false
    modifier lock() {
        require(slot0.unlocked, 'LOK');
        slot0.unlocked = false;
        _;
        slot0.unlocked = true;
    }

ticks & tickBitmap — The liquidity bookkeeping engine
Tick.Info · int24→int16+uint8
    /// @dev Every INITIALIZED tick stores a Tick.Info:
    ///
    ///  liquidityGross  — total liquidity that references this tick as either
    ///                    lower OR upper boundary. Used to know when to
    ///                    un-initialize (set to 0 → delete from bitmap).
    ///
    ///  liquidityNet    — net change to apply to `liquidity` when the current
    ///                    price CROSSES this tick during a swap:
    ///                      +amount when price crosses upward through lowerTick
    ///                      -amount when price crosses upward through upperTick
    ///                    (reversed for downward swaps)
    ///
    ///  feeGrowthOutside{0,1}X128 — fee growth on the OPPOSITE side of this tick
    ///                    from the current price. Used to compute feeGrowthInside.
    ///
    ///  tickCumulativeOutside, secondsPerLiquidityOutsideX128, secondsOutside
    ///                    — oracle values on the outside of this tick.
    mapping(int24 => Tick.Info) public override ticks;

    /// @dev Bitmap of initialized ticks. The full int24 tick space is split:
    ///   wordPos  = tick / 256   (the mapping key, type int16)
    ///   bitPos   = tick % 256   (which bit within the word, type uint8)
    ///
    /// A set bit = an initialized tick. nextInitializedTickWithinOneWord() does
    /// an efficient bit-scan to find the nearest set bit, letting swaps skip
    /// through large uninitiated regions in O(1) per word.
    mapping(int16 => uint256) public override tickBitmap;

observations — On-chain TWAP oracle
circular buffer · 65535 slots · tickCumulative
    /// @dev Fixed-size circular buffer of Oracle.Observation entries.
    /// Each entry records a checkpoint of the cumulative price at a block timestamp.
    ///
    /// Oracle.Observation contains:
    ///   blockTimestamp          — uint32 (wraps every ~136 years)
    ///   tickCumulative          — int56: ∑ tick * Δseconds since pool creation
    ///   secondsPerLiquidityX128 — uint160: ∑ Δseconds/liquidity since creation
    ///   initialized             — bool: true if this slot has been written
    ///
    /// TWAP for a window [T0, T1]:
    ///   avgTick = (tickCumulative[T1] - tickCumulative[T0]) / (T1 - T0)
    ///   avgPrice = 1.0001 ^ avgTick
    ///
    /// By default only slot[0] is active (cardinality = 1). Call
    /// increaseObservationCardinalityNext() to expand — each extra slot
    /// costs ~20,000 gas to initialize (paid by whoever calls the function).
    Oracle.Observation[65535] public override observations;

initialize() — Set starting price; must be called before any swap or mint
sqrtPriceX96 → tick · first observation written
    /// @notice Sets the initial sqrt price and activates the pool.
    /// @param sqrtPriceX96 The initial sqrt(price) in Q64.96 format.
    ///        e.g. to start at price 2000 (USDC per ETH): sqrt(2000) * 2^96
    function initialize(uint160 sqrtPriceX96) external override {
        // Prevent double-initialization
        require(slot0.sqrtPriceX96 == 0, 'AI');  // Already Initialized

        // Derive tick from the provided sqrt price.
        // TickMath.getTickAtSqrtRatio does: floor(log_1.0001(price))
        int24 tick = TickMath.getTickAtSqrtRatio(sqrtPriceX96);

        // Write the first oracle observation (timestamp, tickCumulative=0, spls=0)
        uint16 cardinality;
        uint16 cardinalityNext;
        (cardinality, cardinalityNext) = observations.initialize(_blockTimestamp());

        // Pack everything into slot0.  unlocked = true opens the pool for business.
        slot0 = Slot0({
            sqrtPriceX96               : sqrtPriceX96,
            tick                       : tick,
            observationIndex           : 0,
            observationCardinality     : cardinality,
            observationCardinalityNext : cardinalityNext,
            feeProtocol               : 0,
            unlocked                  : true
        });

        emit Initialize(sqrtPriceX96, tick);
    }

_updatePosition() — Core bookkeeping shared by mint & burn
tick init/destroy · feeGrowthInside · tokensOwed
    /// @dev Private helper. Updates ticks, computes fees owed, mutates the position.
    function _updatePosition(
        address  owner,
        int24    tickLower,
        int24    tickUpper,
        int128   liquidityDelta, // positive = mint, negative = burn
        int24    tick            // current tick from slot0
    ) private returns (Position.Info storage position) {
        position = positions.get(owner, tickLower, tickUpper);

        // Snapshot global fee growth before any state changes
        uint256 _feeGrowthGlobal0X128 = feeGrowthGlobal0X128;
        uint256 _feeGrowthGlobal1X128 = feeGrowthGlobal1X128;

        // Update lower tick if liquidity changed
        bool flippedLower;
        bool flippedUpper;
        if (liquidityDelta != 0) {
            uint32 time = _blockTimestamp();
            (int56 tickCumulative, uint160 secondsPerLiquidityX128) =
                observations.observeSingle(time, 0, slot0.tick, slot0.observationIndex,
                    liquidity, slot0.observationCardinality);

            // ticks.update returns true if the tick just flipped:
            // initialized→uninitialized (liquidityGross hit 0) or vice versa
            flippedLower = ticks.update(
                tickLower, tick, liquidityDelta,
                _feeGrowthGlobal0X128, _feeGrowthGlobal1X128,
                secondsPerLiquidityX128, tickCumulative, time,
                false, /* isUpper = false */
                maxLiquidityPerTick
            );
            flippedUpper = ticks.update(
                tickUpper, tick, liquidityDelta,
                _feeGrowthGlobal0X128, _feeGrowthGlobal1X128,
                secondsPerLiquidityX128, tickCumulative, time,
                true, /* isUpper = true */
                maxLiquidityPerTick
            );

            // If a tick was newly initialized, set its bit in the bitmap.
            // If it was de-initialized (liquidityGross = 0), clear the bit.
            if (flippedLower) tickBitmap.flipTick(tickLower, tickSpacing);
            if (flippedUpper) tickBitmap.flipTick(tickUpper, tickSpacing);
        }

        // Compute fee growth INSIDE the position's tick range.
        // feeGrowthInside = feeGrowthGlobal - feeGrowthBelow(lower) - feeGrowthAbove(upper)
        // Each boundary's "outside" value flips when price crosses it.
        (uint256 feeGrowthInside0X128, uint256 feeGrowthInside1X128) =
            ticks.getFeeGrowthInside(
                tickLower, tickUpper, tick,
                _feeGrowthGlobal0X128, _feeGrowthGlobal1X128
            );

        // Update the position: accumulate new fees into tokensOwed, update snapshot.
        position.update(liquidityDelta, feeGrowthInside0X128, feeGrowthInside1X128);

        // If the position was completely closed and the tick had no other references,
        // clear the tick's storage to recover gas (and get a gas refund).
        if (liquidityDelta < 0) {
            if (flippedLower) ticks.clear(tickLower);
            if (flippedUpper) ticks.clear(tickUpper);
        }
    }

mint() — Add liquidity to a tick range
callback pattern · SqrtPriceMath · balance check
    /// @notice Adds liquidity for msg.sender in [tickLower, tickUpper).
    /// @param recipient   Address that will own the position.
    /// @param tickLower   Lower tick boundary (must be multiple of tickSpacing).
    /// @param tickUpper   Upper tick boundary.
    /// @param amount      Liquidity units to add (NOT token amounts).
    /// @param data        Forwarded to the IUniswapV3MintCallback.
    /// @return amount0    token0 actually deposited.
    /// @return amount1    token1 actually deposited.
    function mint(
        address   recipient,
        int24     tickLower,
        int24     tickUpper,
        uint128   amount,
        bytes calldata data
    ) external override lock noDelegateCall
      returns (uint256 amount0, uint256 amount1)
    {
        require(amount > 0);
        (, int256 amount0Int, int256 amount1Int) =
            _modifyPosition(ModifyPositionParams({
                owner          : recipient,
                tickLower      : tickLower,
                tickUpper      : tickUpper,
                liquidityDelta : int256(amount).toInt128()
            }));

        // _modifyPosition returns signed amounts; for mint they are always positive.
        amount0 = uint256(amount0Int);
        amount1 = uint256(amount1Int);

        // Snapshot balances BEFORE calling the callback.
        // Pattern: optimistically trust the caller, then verify.
        uint256 balance0Before;
        uint256 balance1Before;
        if (amount0 > 0) balance0Before = balance0();
        if (amount1 > 0) balance1Before = balance1();

        // Callback: caller must transfer amount0 and amount1 into this contract.
        // This is where NonfungiblePositionManager (or any router) sends the tokens.
        IUniswapV3MintCallback(msg.sender).uniswapV3MintCallback(
            amount0, amount1, data
        );

        // Verify the callback actually transferred the tokens. This is the
        // safety check that prevents anyone from minting without paying.
        if (amount0 > 0)
            require(balance0() >= balance0Before + amount0, 'M0');
        if (amount1 > 0)
            require(balance1() >= balance1Before + amount1, 'M1');

        emit Mint(msg.sender, recipient, tickLower, tickUpper, amount, amount0, amount1);
    }

collect() — Withdraw accumulated fees from a position
tokensOwed · no callback · capped amounts
    /// @notice Collects tokens owed to a position (fees + withdrawn principal).
    /// @dev Does NOT update feeGrowthInside — call burn(0) first if you want
    ///      to collect fees that accrued since the last update.
    function collect(
        address   recipient,
        int24     tickLower,
        int24     tickUpper,
        uint128   amount0Requested,
        uint128   amount1Requested
    ) external override lock
      returns (uint128 amount0, uint128 amount1)
    {
        // Load position from storage (msg.sender is the position owner).
        Position.Info storage position = positions.get(
            msg.sender, tickLower, tickUpper
        );

        // Cap at what is actually owed — caller may request type(uint128).max to mean "all".
        amount0 = amount0Requested > position.tokensOwed0
            ? position.tokensOwed0
            : amount0Requested;
        amount1 = amount1Requested > position.tokensOwed1
            ? position.tokensOwed1
            : amount1Requested;

        // Decrement owed amounts. Unchecked: amounts are capped above.
        if (amount0 > 0) {
            position.tokensOwed0 -= amount0;
            TransferHelper.safeTransfer(token0, recipient, amount0);
        }
        if (amount1 > 0) {
            position.tokensOwed1 -= amount1;
            TransferHelper.safeTransfer(token1, recipient, amount1);
        }

        emit Collect(msg.sender, recipient, tickLower, tickUpper, amount0, amount1);
    }

burn() — Remove liquidity; credit tokens to position
negative liquidityDelta · collect() to withdraw
    /// @notice Burns liquidity and credits tokens back to the position's tokensOwed.
    /// @dev Tokens are NOT sent here — you must call collect() afterwards.
    ///      Calling burn(0) is a valid no-op that triggers a fee update only.
    function burn(
        int24   tickLower,
        int24   tickUpper,
        uint128 amount
    ) external override lock noDelegateCall
      returns (uint256 amount0, uint256 amount1)
    {
        (Position.Info storage position, int256 amount0Int, int256 amount1Int) =
            _modifyPosition(ModifyPositionParams({
                owner          : msg.sender,
                tickLower      : tickLower,
                tickUpper      : tickUpper,
                liquidityDelta : -int256(amount).toInt128()
            }));

        // Negate because _modifyPosition returns negative values for burn.
        amount0 = uint256(-amount0Int);
        amount1 = uint256(-amount1Int);

        // Credit the owed amounts into the position so collect() can withdraw them.
        // tokensOwed is uint128, so safe-cast from uint256.
        if (amount0 > 0 || amount1 > 0) {
            (position.tokensOwed0, position.tokensOwed1) = (
                position.tokensOwed0 + uint128(amount0),
                position.tokensOwed1 + uint128(amount1)
            );
        }
        emit Burn(msg.sender, tickLower, tickUpper, amount, amount0, amount1);
    }

swap() — The core function. Step through ticks, compute amounts, call back
SwapMath · tick crossing · oracle write · callback
    /// @notice Swap token0 for token1, or token1 for token0.
    /// @param recipient      Receives the output tokens.
    /// @param zeroForOne     true = swap token0→token1 (price goes DOWN).
    /// @param amountSpecified Positive = exact input, Negative = exact output.
    /// @param sqrtPriceLimitX96  Price cannot move past this limit. Acts as slippage guard.
    /// @param data           Forwarded to IUniswapV3SwapCallback.
    function swap(
        address   recipient,
        bool      zeroForOne,
        int256    amountSpecified,
        uint160   sqrtPriceLimitX96,
        bytes calldata data
    ) external override lock noDelegateCall
      returns (int256 amount0, int256 amount1)
    {
        require(amountSpecified != 0, 'AS');

        // Cache slot0 to avoid repeated SLOADs inside the loop.
        Slot0 memory slot0Start = slot0;

        // Validate price limit: must be between current price and the appropriate extreme.
        require(
            zeroForOne
                ? sqrtPriceLimitX96 < slot0Start.sqrtPriceX96
                    && sqrtPriceLimitX96 > TickMath.MIN_SQRT_RATIO
                : sqrtPriceLimitX96 > slot0Start.sqrtPriceX96
                    && sqrtPriceLimitX96 < TickMath.MAX_SQRT_RATIO,
            'SPL'
        );

        // Cache fee protocol split to local var.
        bool exactInput = amountSpecified > 0;

        // ── Swap state: tracks progress across tick-crossing iterations ────────
        struct SwapState {
            int256  amountSpecifiedRemaining; // how much is left to fill
            int256  amountCalculated;         // cumulative output (or input for exactOutput)
            uint160 sqrtPriceX96;             // current sqrt price (moves during swap)
            int24   tick;                     // current tick
            uint256 feeGrowthGlobalX128;      // accumulates fee growth for the swap direction
            uint128 protocolFee;              // protocol's cut so far
            uint128 liquidity;                // current active liquidity
        }

        SwapState memory state = SwapState({
            amountSpecifiedRemaining: amountSpecified,
            amountCalculated        : 0,
            sqrtPriceX96           : slot0Start.sqrtPriceX96,
            tick                   : slot0Start.tick,
            feeGrowthGlobalX128    : zeroForOne ? feeGrowthGlobal0X128 : feeGrowthGlobal1X128,
            protocolFee            : 0,
            liquidity              : liquidity
        });

        // ═══════════════════════════════════════════════════════════════════
        // THE MAIN LOOP — steps through one tick at a time until the swap
        // is filled, the price limit is hit, or no more ticks exist.
        // ═══════════════════════════════════════════════════════════════════
        while (
            state.amountSpecifiedRemaining != 0
            && state.sqrtPriceX96 != sqrtPriceLimitX96
        ) {
            struct StepComputations {
                uint160 sqrtPriceStartX96;  // price at step start
                int24   tickNext;            // next initialized tick in swap direction
                bool    initialized;         // whether tickNext is actually initialized
                uint160 sqrtPriceNextX96;   // sqrt price at tickNext
                uint256 amountIn;            // input consumed in this step
                uint256 amountOut;           // output generated in this step
                uint256 feeAmount;           // fee in this step
            }
            StepComputations memory step;

            step.sqrtPriceStartX96 = state.sqrtPriceX96;

            // ── 1. Find the next initialized tick in the swap direction ────────
            // TickBitmap.nextInitializedTickWithinOneWord() does a masked bit-scan
            // on a single 256-bit word, returning the nearest set bit.
            (step.tickNext, step.initialized) =
                tickBitmap.nextInitializedTickWithinOneWord(
                    state.tick, tickSpacing, zeroForOne
                );

            // Clamp to valid range (avoid extreme ticks that have no price)
            if (step.tickNext < TickMath.MIN_TICK) step.tickNext = TickMath.MIN_TICK;
            if (step.tickNext > TickMath.MAX_TICK) step.tickNext = TickMath.MAX_TICK;

            // Convert tick to sqrt price for SwapMath
            step.sqrtPriceNextX96 = TickMath.getSqrtRatioAtTick(step.tickNext);

            // ── 2. Compute how much can be swapped before hitting tickNext ─────
            // SwapMath.computeSwapStep returns the new sqrt price, amounts, and fee.
            // It clamps to sqrtPriceLimitX96 if that is hit first.
            (
                state.sqrtPriceX96,
                step.amountIn,
                step.amountOut,
                step.feeAmount
            ) = SwapMath.computeSwapStep(
                state.sqrtPriceX96,
                (zeroForOne
                    ? step.sqrtPriceNextX96 < sqrtPriceLimitX96
                    : step.sqrtPriceNextX96 > sqrtPriceLimitX96)
                    ? sqrtPriceLimitX96
                    : step.sqrtPriceNextX96,
                state.liquidity,
                state.amountSpecifiedRemaining,
                fee
            );

            // ── 3. Update remaining amount and fee growth ──────────────────────
            if (exactInput) {
                state.amountSpecifiedRemaining -= (step.amountIn + step.feeAmount).toInt256();
                state.amountCalculated = state.amountCalculated.sub(step.amountOut.toInt256());
            } else {
                state.amountSpecifiedRemaining += step.amountOut.toInt256();
                state.amountCalculated = state.amountCalculated.add(
                    (step.amountIn + step.feeAmount).toInt256()
                );
            }

            // Split fee: protocol portion goes to protocolFees, rest to LP fee growth.
            uint256 feeProtocol = zeroForOne
                ? slot0Start.feeProtocol % 16
                : slot0Start.feeProtocol >> 4;

            if (feeProtocol > 0) {
                uint256 delta = step.feeAmount / feeProtocol;
                step.feeAmount -= delta;
                state.protocolFee += uint128(delta);
            }

            // Accumulate per-unit-liquidity fee growth for LP positions.
            if (state.liquidity > 0)
                state.feeGrowthGlobalX128 +=
                    FullMath.mulDiv(step.feeAmount, FixedPoint128.Q128, state.liquidity);

            // ── 4. Cross the tick if we reached it exactly ─────────────────────
            if (state.sqrtPriceX96 == step.sqrtPriceNextX96) {
                if (step.initialized) {
                    // Write an oracle observation before modifying liquidity.
                    (int56 tickCumulative, uint160 secondsPerLiquidityX128) =
                        observations.observeSingle(
                            _blockTimestamp(), 0,
                            slot0Start.tick,
                            slot0Start.observationIndex,
                            state.liquidity,
                            slot0Start.observationCardinality
                        );

                    // Cross the tick: flip its feeGrowthOutside values.
                    // Returns liquidityNet: the net change to apply to state.liquidity.
                    int128 liquidityNet = ticks.cross(
                        step.tickNext,
                        (zeroForOne ? state.feeGrowthGlobalX128 : feeGrowthGlobal0X128),
                        (zeroForOne ? feeGrowthGlobal1X128 : state.feeGrowthGlobalX128),
                        secondsPerLiquidityX128,
                        tickCumulative,
                        _blockTimestamp()
                    );

                    // For downward swaps, liquidityNet is negated.
                    // (Liquidity added when entering a range from above = removed from below)
                    if (zeroForOne) liquidityNet = -liquidityNet;

                    state.liquidity = liquidityNet < 0
                        ? state.liquidity - uint128(-liquidityNet)
                        : state.liquidity + uint128(liquidityNet);
                }

                // Adjust tick: we crossed into the next range.
                // Subtract 1 for zeroForOne because tick is the LOWER boundary.
                state.tick = zeroForOne ? step.tickNext - 1 : step.tickNext;
            } else if (state.sqrtPriceX96 != step.sqrtPriceStartX96) {
                // Price moved but did not reach the next tick — recompute tick from price.
                state.tick = TickMath.getTickAtSqrtRatio(state.sqrtPriceX96);
            }
        } // end while

        // ── 5. Write final state back to storage ───────────────────────────────
        if (state.tick != slot0Start.tick) {
            // Write an oracle observation for this swap (at most once per block).
            uint16 observationIndex;
            uint16 observationCardinality;
            (observationIndex, observationCardinality) = observations.write(
                slot0Start.observationIndex,
                _blockTimestamp(),
                slot0Start.tick,
                state.liquidity,
                slot0Start.observationCardinality,
                slot0Start.observationCardinalityNext
            );
            // Single SSTORE: update price, tick, and oracle index atomically.
            (slot0.sqrtPriceX96, slot0.tick, slot0.observationIndex, slot0.observationCardinality) =
                (state.sqrtPriceX96, state.tick, observationIndex, observationCardinality);
        } else {
            // Price didn't change tick — just update sqrtPriceX96.
            slot0.sqrtPriceX96 = state.sqrtPriceX96;
        }

        if (liquidity != state.liquidity) liquidity = state.liquidity;

        // Write fee growth accumulators and protocol fees.
        if (zeroForOne) {
            feeGrowthGlobal0X128 = state.feeGrowthGlobalX128;
            if (state.protocolFee > 0) protocolFees.token0 += state.protocolFee;
        } else {
            feeGrowthGlobal1X128 = state.feeGrowthGlobalX128;
            if (state.protocolFee > 0) protocolFees.token1 += state.protocolFee;
        }

        // ── 6. Determine final signed amounts and call the swap callback ───────
        (amount0, amount1) = zeroForOne == exactInput
            ? (amountSpecified - state.amountSpecifiedRemaining, state.amountCalculated)
            : (state.amountCalculated, amountSpecified - state.amountSpecifiedRemaining);

        // Transfer output to recipient, then call the callback to collect input.
        if (zeroForOne) {
            if (amount1 < 0)
                TransferHelper.safeTransfer(token1, recipient, uint256(-amount1));
            uint256 balance0Before = balance0();
            IUniswapV3SwapCallback(msg.sender).uniswapV3SwapCallback(amount0, amount1, data);
            require(balance0() >= balance0Before + uint256(amount0), 'IIA');
        } else {
            if (amount0 < 0)
                TransferHelper.safeTransfer(token0, recipient, uint256(-amount0));
            uint256 balance1Before = balance1();
            IUniswapV3SwapCallback(msg.sender).uniswapV3SwapCallback(amount0, amount1, data);
            require(balance1() >= balance1Before + uint256(amount1), 'IIA');
        }

        emit Swap(msg.sender, recipient, amount0, amount1,
            state.sqrtPriceX96, state.liquidity, state.tick);
    }

    /// @notice Lend up to the pool's full balance of token0 and/or token1.
    /// The caller must repay the borrowed amount PLUS a fee within the same transaction.
    function flash(
        address   recipient,
        uint256   amount0,    // token0 to borrow
        uint256   amount1,    // token1 to borrow
        bytes calldata data
    ) external override lock noDelegateCall {
        uint128 _liquidity = liquidity;
        require(_liquidity > 0, 'L');

        // Compute fee: same rate as the swap fee.
        // fee1 = ceil(amount1 * fee / 1_000_000)
        uint256 fee0 = FullMath.mulDivRoundingUp(amount0, fee, 1e6);
        uint256 fee1 = FullMath.mulDivRoundingUp(amount1, fee, 1e6);

        // Record balances BEFORE sending tokens out.
        uint256 balance0Before = balance0();
        uint256 balance1Before = balance1();

        // Transfer the requested amounts to the recipient.
        if (amount0 > 0) TransferHelper.safeTransfer(token0, recipient, amount0);
        if (amount1 > 0) TransferHelper.safeTransfer(token1, recipient, amount1);

        // Recipient does its arbitrage / liquidation / whatever in the callback.
        IUniswapV3FlashCallback(msg.sender).uniswapV3FlashCallback(fee0, fee1, data);

        // After callback: verify full repayment including fees.
        require(balance0() >= balance0Before + fee0, 'F0');
        require(balance1() >= balance1Before + fee1, 'F1');

        // Distribute flash fees to LP fee growth (and protocol if enabled).
        uint256 paid0 = balance0() - balance0Before;
        uint256 paid1 = balance1() - balance1Before;
        // [fee distribution logic omitted for brevity — mirrors swap fee logic]

        emit Flash(msg.sender, recipient, amount0, amount1, paid0, paid1);
    }

snapshotCumulativesInside() — Oracle data for a specific tick range
in-range TWAP · tickCumulativeInside · for incentives
    /// @notice Returns cumulative values INSIDE a given tick range.
    /// Used by yield farming contracts to reward LPs proportionally to time-in-range.
    ///
    /// The inside values are computed from the "outside" accumulators stored
    /// on the lower and upper ticks, using the current tick to determine which
    /// side is which.
    function snapshotCumulativesInside(int24 tickLower, int24 tickUpper)
        external view override
        returns (
            int56   tickCumulativeInside,
            uint160 secondsPerLiquidityInsideX128,
            uint32  secondsInside
        )
    {
        checkTicks(tickLower, tickUpper);
        Tick.Info storage lower = ticks[tickLower];
        Tick.Info storage upper = ticks[tickUpper];
        int24 _tick = slot0.tick;

        // "Outside" means on the opposite side of the tick from the current price.
        // For ticks BELOW current price: outside = below the tick = NOT inside.
        // For ticks ABOVE current price: outside = above the tick = NOT inside.
        // inside = global - below - above

        int56   lowerTickCumulative       = lower.tickCumulativeOutside;
        uint160 lowerSecondsPerLiquidity = lower.secondsPerLiquidityOutsideX128;
        uint32  lowerSeconds             = lower.secondsOutside;

        if (_tick < tickLower) {
            // Current price is below the range; the lowerTick's "outside" is actually inside.
            (int56 tc, uint160 spl, uint32 s) = _currentObservation();
            lowerTickCumulative       = tc       - lowerTickCumulative;
            lowerSecondsPerLiquidity = spl      - lowerSecondsPerLiquidity;
            lowerSeconds             = s        - lowerSeconds;
        }

        // [mirror logic for upper tick — omitted for brevity]

        tickCumulativeInside             = upperTickCumulative - lowerTickCumulative;
        secondsPerLiquidityInsideX128   = upperSecondsPerLiquidity - lowerSecondsPerLiquidity;
        secondsInside                   = upperSeconds - lowerSeconds;
    }

setFeeProtocol() & collectProtocol() — Factory owner governance

    /// @notice Enables or changes the protocol fee split. Only the factory owner.
    /// @param feeProtocol0 Protocol's share of token0 fees (0 = disabled, 4–10 valid).
    ///        Value N means LP gets (N-1)/N, protocol gets 1/N.
    ///        e.g. feeProtocol0=6 → protocol takes 1/6 of swap fees in token0.
    function setFeeProtocol(uint8 feeProtocol0, uint8 feeProtocol1)
        external override lock onlyFactoryOwner
    {
        require(
            (feeProtocol0 == 0 || (feeProtocol0 >= 4 && feeProtocol0 <= 10)) &&
            (feeProtocol1 == 0 || (feeProtocol1 >= 4 && feeProtocol1 <= 10))
        );
        uint8 feeProtocolOld = slot0.feeProtocol;
        // Pack both values into one byte: low nibble = token0, high nibble = token1.
        slot0.feeProtocol = feeProtocol0 + (feeProtocol1 << 4);
        emit SetFeeProtocol(feeProtocolOld % 16, feeProtocolOld >> 4, feeProtocol0, feeProtocol1);
    }

collectProtocol() — Withdraw accrued protocol fees
factory owner only · transfers from protocolFees
    /// @notice Withdraws protocol fees to the recipient. Factory owner only.
    function collectProtocol(
        address   recipient,
        uint128   amount0Requested,
        uint128   amount1Requested
    ) external override lock onlyFactoryOwner
      returns (uint128 amount0, uint128 amount1)
    {
        amount0 = amount0Requested > protocolFees.token0
            ? protocolFees.token0
            : amount0Requested;
        amount1 = amount1Requested > protocolFees.token1
            ? protocolFees.token1
            : amount1Requested;

        if (amount0 > 0) {
            protocolFees.token0 -= amount0;
            TransferHelper.safeTransfer(token0, recipient, amount0);
        }
        if (amount1 > 0) {
            protocolFees.token1 -= amount1;
            TransferHelper.safeTransfer(token1, recipient, amount1);
        }
        emit CollectProtocol(msg.sender, recipient, amount0, amount1);
    }
